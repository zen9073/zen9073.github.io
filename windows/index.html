    在此处打开命令窗口

    Windows Registry Editor Version 5.00

[HKEY_CLASSES_ROOT\Directory\background\shell\OpenCMDHere]
@="在此处打开命令窗口"
"Icon"="cmd.exe"
"ShowBasedOnVelocityId"=dword:00639bc8

[HKEY_CLASSES_ROOT\Directory\background\shell\OpenCMDHere\command]
@="cmd.exe /s /k pushd \"%V\""
